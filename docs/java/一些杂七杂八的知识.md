//两种写法效率是不同的：
int[10][128] arr;
    
for (i = 0; i < 10; i++)
        for (j = 0; j < 128; j++);
                arr[i][j] = 1;

for (i = 0; i < 128; i++)
        for (j = 0; j < 10; j++);
                arr[j][i] = 1;
